112real_t Problem::calc_ŷ_dᵀŷ(rvec g_ŷ, crvec y, crvec Σ) const {
113 if (Σ.size() == 1) {
114 // ζ = g(x) + Σ⁻¹y
115 g_ŷ += (1 / Σ(0)) * y;
116 // d = ζ - Π(ζ, D)
117 g_ŷ = projecting_difference(g_ŷ, D);
118 // dᵀŷ, ŷ = Σ d
119 real_t dᵀŷ = Σ(0) * g_ŷ.dot(g_ŷ);
120 g_ŷ *= Σ(0);
121 return dᵀŷ;
122 } else {
123 // ζ = g(x) + Σ⁻¹y
124 g_ŷ += Σ.asDiagonal().inverse() * y;
125 // d = ζ - Π(ζ, D)
126 g_ŷ = projecting_difference(g_ŷ, D);
127 // dᵀŷ, ŷ = Σ d
128 real_t dᵀŷ = 0;
129 for (unsigned i = 0; i < m; ++i) {
130 dᵀŷ += g_ŷ(i) * Σ(i) * g_ŷ(i); // TODO: vectorize
131 g_ŷ(i) = Σ(i) * g_ŷ(i);
132 }
133 return dᵀŷ;
134 }
