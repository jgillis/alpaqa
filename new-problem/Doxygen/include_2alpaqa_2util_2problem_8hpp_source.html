19/**
20 * @class Problem
21 * @brief Problem description for minimization problems.
22 *
23 * @f[ \begin{aligned}
24 * & \underset{x}{\text{minimize}}
25 * & & f(x) &&&& f : \mathbb{R}^n \rightarrow \mathbb{R} \\
26 * & \text{subject to}
27 * & & \underline{x} \le x \le \overline{x} \\
28 * &&& \underline{z} \le g(x) \le \overline{z} &&&& g :
29 * \mathbb{R}^n \rightarrow \mathbb{R}^m
30 * \end{aligned} @f]
31 */
32struct Problem {
33 /// Number of decision variables, dimension of x
34 unsigned int n;
35 /// Number of constraints, dimension of g(x) and z
36 unsigned int m;
37 /// Constraints of the decision variables, @f$ x \in C @f$
38 Box C{vec::Constant(n, +inf), vec::Constant(n, -inf)};
39 /// Other constraints, @f$ g(x) \in D @f$
40 Box D{vec::Constant(m, +inf), vec::Constant(m, -inf)};
41
42 Problem(unsigned int n, unsigned int m) : n(n), m(m) {}
43 Problem(unsigned int n, unsigned int m, Box C, Box D)
44 : n(n), m(m), C{std::move(C)}, D{std::move(D)} {}
45
46 Problem(const Problem &) = default;
47 Problem &operator=(const Problem &) = default;
48 Problem(Problem &&) = default;
49 Problem &operator=(Problem &&) = default;

51 virtual std::unique_ptr<Problem> clone() const &;
52 virtual std::unique_ptr<Problem> clone() &&;
53 virtual ~Problem() = default;

85 /// @f$ \nabla g_i(x) @f$

138 /// @{
139

141 /// ∇ψ.
142 /// @f[ \psi(x) = f(x) + \tfrac{1}{2}
143 /// \text{dist}_\Sigma^2\left(g(x) + \Sigma^{-1}y,\;D\right) @f]

145 /// + \Sigma^{-1}y\right)\right) @f]

147 crvec y, ///< [in] Lagrange multipliers @f$ y @f$

149 rvec ŷ ///< [out] @f$ \hat y @f$
150 ) const;

155 rvec grad_ψ, ///< [out] @f$ \nabla \psi(x) @f$
156 rvec work_n ///< Dimension @f$ n @f$
157 ) const;

159 /// @f[ \nabla \psi(x) = \nabla f(x) + \nabla g(x)\,\hat y(x) @f]
160 virtual void eval_grad_ψ(crvec x, ///< [in] Decision variable @f$ x @f$
161 crvec y, ///< [in] Lagrange multipliers @f$ y @f$
162 crvec Σ, ///< [in] Penalty weights @f$ \Sigma @f$
163 rvec grad_ψ, ///< [out] @f$ \nabla \psi(x) @f$
164 rvec work_n, ///< Dimension @f$ n @f$
165 rvec work_m ///< Dimension @f$ m @f$
166 ) const;

168 /// @f[ \psi(x) = f(x) + \tfrac{1}{2}
169 /// \text{dist}_\Sigma^2\left(g(x) + \Sigma^{-1}y,\;D\right) @f]

171 virtual real_t
172 eval_ψ_grad_ψ(crvec x, ///< [in] Decision variable @f$ x @f$
173 crvec y, ///< [in] Lagrange multipliers @f$ y @f$
174 crvec Σ, ///< [in] Penalty weights @f$ \Sigma @f$
175 rvec grad_ψ, ///< [out] @f$ \nabla \psi(x) @f$
176 rvec work_n, ///< Dimension @f$ n @f$
177 rvec work_m ///< Dimension @f$ m @f$
178 ) const;

181
182 /// @name Helpers
183 /// @{
184

194 real_t calc_ŷ_dᵀŷ(rvec g_ŷ, crvec y, crvec Σ) const;
195
196 /// @}

459 template <class TimeT, class FunT>
460 static auto timed(TimeT &time, const FunT &f) -> decltype(f()) {
461 if constexpr (std::is_same_v<decltype(f()), void>) {
462 auto t0 = std::chrono::steady_clock::now();
463 f();
464 auto t1 = std::chrono::steady_clock::now();
465 time += t1 - t0;
466 } else {
467 auto t0 = std::chrono::steady_clock::now();
468 auto res = f();
469 auto t1 = std::chrono::steady_clock::now();
470 time += t1 - t0;
471 return res;
472 }
