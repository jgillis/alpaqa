61 auto load_g_unknown_dims =

63 casadi::Function gfun = casadi::external("g", so_name);
64 using std::operator""s;
65 if (gfun.n_in() != 2)
66 throw std::invalid_argument(
67 "Invalid number of input arguments: got "s +
68 std::to_string(gfun.n_in()) + ", should be 2.");
69 if (gfun.n_out() > 1)
70 throw std::invalid_argument(
71 "Invalid number of output arguments: got "s +
72 std::to_string(gfun.n_in()) + ", should be 0 or 1.");
73 if (gfun.size2_in(0) != 1)
74 throw std::invalid_argument(
75 "First input argument should be a column vector.");
76 if (gfun.size2_in(1) != 1)
77 throw std::invalid_argument(
78 "Second input argument should be a column vector.");
79 if (gfun.n_out() == 1 && gfun.size2_out(0) != 1)
80 throw std::invalid_argument(
81 "First output argument should be a column vector.");
82 if (n == 0)
83 n = gfun.size1_in(0);
84 if (m == 0 && gfun.n_out() == 1)
85 m = gfun.size1_out(0);
86 if (p == 0)
87 p = gfun.size1_in(1);
88 if (gfun.n_out() == 0) {
89 if (m != 0)
90 throw std::invalid_argument(
91 "Function g has no outputs but m != 0");
92 return std::nullopt;
93 }
94 CasADiFunctionEvaluator<2, 1> g{std::move(gfun)};
95 g.validate_dimensions({dim(n, 1), dim(p, 1)}, {dim(m, 1)});
96 return std::make_optional(std::move(g));
97 };
